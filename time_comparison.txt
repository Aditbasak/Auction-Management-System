SELECT a.auction_id as Auction_Id, p.product_id as Product_Id, p.p_name AS Name, p.p_desc AS Description, p.baseprice as BasePrice, a.current_bid as CurrentBid FROM products p INNER JOIN auctions a ON p.product_id = a.product WHERE (a.bid_start <= TO_TIMESTAMP('11-04-2024 11:50', 'DD-MM-YYYY HH24:MI:SS')) AND (a.bid_end > TO_TIMESTAMP('11-04-2024 11:50', 'DD-MM-YYYY HH24:MI:SS'));
